def calculate_slippage(price, quantity, market_depth, volatility):
    """
    Base slippage: 0.05%
    Volume impact: qty / avg_volume * 0.5%
    Volatility impact: ATR% * 0.02
    """
    base = 0.0005
    volume_impact = (quantity / market_depth.avg_volume_5min) * 0.005
    vol_impact = volatility * 0.02

    total_slippage = base + volume_impact + vol_impact
    return price * (1 + total_slippage)  # for buys

def walk_forward_backtest(data, train_window=365, test_window=90):
    """
    Rolling window backtest:
    - Train –Ω–∞ N –¥–Ω–µ–π

    - –î–≤–∏–≥–∞—Ç—å –æ–∫–Ω–æ –≤–ø–µ—Ä—ë–¥
    """
    results = []

    for start in range(0, len(data) - train_window - test_window, test_window):
        train_data = data[start:start + train_window]
        test_data = data[start + train_window:start + train_window + test_window]

        # Optimize parameters on train

        # Test with those parameters

#### 8. **Real Kelly Criterion**

```python
class KellyCalculator:
    def __init__(self, lookback=50):
        self.trade_history = []
        self.lookback = lookback

    def update(self, trade_pnl):
        self.trade_history.append(trade_pnl)
        if len(self.trade_history) > self.lookback:
            self.trade_history.pop(0)

    def calculate_kelly(self):
        wins = [x for x in self.trade_history if x > 0]
        losses = [x for x in self.trade_history if x < 0]

        if not wins or not losses:
            return 0.01  # Conservative fallback

        win_rate = len(wins) / len(self.trade_history)
        avg_win = np.mean(wins)
        avg_loss = abs(np.mean(losses))

        # Kelly formula
        kelly = (win_rate * avg_win - (1 - win_rate) * avg_loss) / avg_win

        # Fractional Kelly (0.25 –¥–ª—è –∫–æ–Ω—Å–µ—Ä–≤–∞—Ç–∏–≤–Ω–æ—Å—Ç–∏)
        return max(0.001, min(0.05, kelly * 0.25))
```
